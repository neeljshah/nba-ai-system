    ```python
    """Abstract base for all NBA AI sklearn models.

    Every model must implement fit() and predict(). Models are persisted
    with joblib to models/artifacts/{model_name}.joblib.
    """

    import pathlib
    from abc import ABC, abstractmethod

    import joblib

    ARTIFACTS_DIR = pathlib.Path(__file__).parent / "artifacts"


    class BaseModel(ABC):
        """Minimal contract every Phase 3 model must satisfy.

        Subclasses are expected to wrap a sklearn estimator and expose
        a predict() method returning Python native types (float, dict, list)
        so Phase 4 API can JSON-serialize results without extra conversion.
        """

        model_name: str  # override in subclass, used as artifact filename

        @abstractmethod
        def fit(self, df) -> "BaseModel":
            """Train on a pandas DataFrame. Returns self for chaining."""

        @abstractmethod
        def predict(self, features: dict) -> object:
            """Score a single observation dict. Returns JSON-serializable value."""

        def save(self) -> pathlib.Path:
            """Serialize model to models/artifacts/{model_name}.joblib."""
            ARTIFACTS_DIR.mkdir(parents=True, exist_ok=True)
            path = ARTIFACTS_DIR / f"{self.model_name}.joblib"
            joblib.dump(self, path)
            return path

        @classmethod
        def load(cls, model_name: str) -> "BaseModel":
            """Deserialize model from models/artifacts/{model_name}.joblib."""
            path = ARTIFACTS_DIR / f"{model_name}.joblib"
            if not path.exists():
                raise FileNotFoundError(
                    f"No artifact at {path}. Run training first."
                )
            return joblib.load(path)
    ```

    Create models/training_data.py. Each function takes an optional `game_id: str | None = None`
    to scope queries (None = all games = full training set) and returns a pandas DataFrame.
    Use `pd.read_sql(sql, conn)` pattern. Import get_connection from tracking.database.

    Functions to implement:

    **load_shot_data(game_id=None) -> pd.DataFrame**
    Query: JOIN shot_logs sl WITH feature_vectors fv ON
           sl.game_id = fv.game_id AND sl.frame_number = fv.frame_number
    Columns: sl.x, sl.y, sl.made (bool), fv.nearest_defender_dist,
             fv.closing_speed, sl.game_id, sl.track_id
    Derived columns (compute in Python after query):
    - shot_angle: abs(atan2(y - 240, x - 460)) in degrees  [basket centre approx (460,240)]
    - court_zone: 'paint' if x<120 or x>800 else 'midrange' if x<200 or x>720 else 'three'
    - fatigue_proxy: closing_speed * -1  (negative closing speed = defender closing = more fatigue)

    **load_possession_data(game_id=None) -> pd.DataFrame**
    Query: feature_vectors LEFT JOIN momentum_snapshots ms ON fv.game_id = ms.game_id
           (match segment: segment_id = frame_number // 150 — 5 seconds at 30fps)
    Columns: fv.game_id, fv.possession_id, fv.frame_number,
             fv.convex_hull_area, fv.avg_inter_player_dist,
             ms.scoring_run, ms.possession_streak, ms.swing_point
    Add result column 'won' placeholder as 0 (Phase 3 trains on synthetic patterns;
    real win labels require play-by-play data deferred to v2).

    **load_momentum_data(game_id=None) -> pd.DataFrame**
    Query: momentum_snapshots
    Columns: game_id, segment_id, scoring_run, possession_streak,
             swing_point (as int 0/1 — the label), timestamp_ms
    Derived: lag features — scoring_run_prev = scoring_run.shift(1) within game_id group,
             streak_delta = possession_streak - possession_streak.shift(1)
             (compute after loading with pandas groupby/shift)

    **load_player_event_data(game_id=None) -> pd.DataFrame**
    Query: JOIN detected_events de WITH shot_logs sl ON
           de.game_id = sl.game_id AND de.track_id = sl.track_id
    Columns: de.track_id, de.game_id, de.event_type, de.confidence,
             sl.made (as int), sl.x, sl.y
    Used for player impact — group by track_id to compute shot made rate, event mix.

    **load_lineup_data(game_id=None) -> pd.DataFrame**
    Query: tracking_coordinates tc WHERE object_type = 'player'
    Columns: game_id, frame_number, track_id
    Purpose: reconstruct which players were on-court each frame.
    Aggregate: for each (game_id, frame_number), collect set of track_ids as lineup tuple.
    Return one row per (game_id, lineup_key) where lineup_key = sorted tuple of 5 track_ids.
    (Lineup scoring model uses EPA from player impact — this provides the roster-frame join.)

    All functions must:
    - Open and close connection inside the function (no leaked connections)
    - Return empty DataFrame with correct columns if no rows found (graceful)
    - Log a warning via logging.getLogger(__name__) if result is empty
